 
public class Solution
{
    
    public static Node deleteLastNode(Node head) {
        
        // Check if the list is empty
        if(head == null) 
            return head;
        
        // Check if the list has only one node
        if(head.next == null) 
            return null;

        // Create a temporary pointer to traverse the list
        Node temp = head;

        // Traverse the list until the second-to-last node
        while(temp.next.next != null){
            temp = temp.next;
        }

        // Store the reference to the last node
        Node lastNode = temp.next;

        // Remove the reference to the last node, effectively deleting it
        temp.next = null;

        // Set the last node reference to null
        lastNode = null;

        // Return the head of the doubly linked list after deletion
        return head;
    }
}
