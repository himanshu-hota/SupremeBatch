// Method 1

class Solution {
    public void deleteNode(ListNode node) {
        
        // Next ko pakdo.
        ListNode next1 = node.next;
        
        // uske bhi next ko pakddo.
        ListNode next2 = next1.next; // or node.next.next

        // given node ka data iske next k data se update kar do.
        node.val = next1.val;

        // node ka next ko second wale next pe point kar do.
        node.next = next2;

        // first wale next ko null kar do.
        next1 = null;

    }
}

// Method 2

class Solution {
    public void deleteNode(ListNode node) {
        
        ListNode temp = node;

        ListNode prev = null;

        while(temp.next != null){
            
            prev = temp;
            temp.val = temp.next.val;
            temp = temp.next;
            
        }

        prev.next = null;
    }
}
