  public boolean hasCycle(ListNode head) {

    if(head == null || head.next == null) return false;

     ListNode slow = head;
     ListNode fast = head; 
     // loop till the fast reaches the end
     while(fast != null && fast.next != null){
         slow = slow.next;
         fast = fast.next.next;
        
         // if any point of time slow and fast both are at same node then
         // they are standing at any node that is in the path of the cycle
         if(slow == fast) return true;
     }

     return false;
    }


public class Solution {
   
    public boolean hasCycle(ListNode head) {

    if(head == null || head.next == null) return false;

     ListNode slow = head;
     ListNode fast = head.next; // yaha galti karoge, = head mat karna.

     while(fast != null){
         if(slow == fast) return true;

         fast = fast.next;
         if(fast != null){
             fast = fast.next;
             slow = slow.next;
         }
     }

     return false;
    }
}
