Node copyList(Node head) {
    // Check if the list is empty
    if (head == null) return head;

    // Step 1: Clone nodes
    Node curr = head;
    while (curr != null) {
        Node clone = new Node(curr.data); // Create a clone node with the same data
        clone.next = curr.next; // Set the next of the clone node to the next of the original node
        curr.next = clone; // Set the next of the original node to the clone node
        curr = clone.next; // Move to the next original node
    }

    // Step 2: Assign random
    curr = head;
    while (curr != null) {
        Node clone = curr.next; // Get the clone node corresponding to the original node
        clone.arb = curr.arb != null ? curr.arb.next : null; // Set the arbitrary pointer of the clone node based on the original node's arbitrary pointer
        curr = curr.next.next; // Move to the next original node
    }

    // Step 3: Detach clone and original
    curr = head;
    Node cloneHead = head.next; // Get the head of the cloned list
    while (curr != null) {
        Node clonedNode = curr.next; // Get the cloned node
        curr.next = curr.next.next; // Set the next of the original node to the next of the cloned node
        curr = curr.next; // Move to the next original node
        clonedNode.next = clonedNode.next != null ? clonedNode.next.next : null; // Set the next of the cloned node to the next of the next cloned node (or null)
    }

    return cloneHead; // Return the head of the cloned list
}
