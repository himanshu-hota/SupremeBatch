 public ListNode middleNode(ListNode head) {
        if(head == null) return null;

        if(head.next == null) return head;

        ListNode slow = head;
        ListNode fast = head;
        
        while(slow != null && fast != null){
            fast = fast.next;
            if(fast != null){
                fast = fast.next;
                slow = slow.next;
            }
        }

        return slow;
    }
