
public class Solution {

    // Method to construct a linked list from an array of integers.
    public static Node constructLL(int[] arr) {

        // If the array is empty, return a new Node with default values.
        if (arr.length == 0) return new Node();

        // Create the head node with the first element of the array.
        Node head = new Node(arr[0]);

        // Temporary node to traverse the linked list.
        Node temp = head;

        // Iterate through the array to create and link nodes in the linked list.
        for (int i = 1; i < arr.length; i++) {
            Node curr = new Node(arr[i]);
            temp.next = curr;
            temp = temp.next;
        }

        // Return the head of the linked list.
        return head;
    }
}
