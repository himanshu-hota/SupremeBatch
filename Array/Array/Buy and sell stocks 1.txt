	public static void BuyAndSellStocks1(int arr[]) {
		int profit = 0;
		int buy= arr[0];
		
		for(int i = 1 ;i<arr.length; i++) {
			if(buy>arr[i]) {
				buy = arr[i];
			}else if(arr[i] - buy > profit) {
				profit = arr[i] - buy;
			}
		}
				
		System.out.println("Maximum profit is : " + profit);
	}
