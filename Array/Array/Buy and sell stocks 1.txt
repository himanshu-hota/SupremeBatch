		// initially mark min price as todays price
        int minPrice = prices[0];
		// initially mark max profit to MIN.
        int maxProfit = Integer.MIN_VALUE;


        for(int i = 0; i<prices.length ;i++){
			// Calculate profit if we sell today
            int todaysProfit = prices[i] - minPrice ;
			
			// update(sell) profit if we found higher profit today
            if(todaysProfit > maxProfit) maxProfit = todaysProfit;

			// update(buy) minPrice if we found a price drop today 
            if(prices[i] < minPrice) minPrice = prices[i];
        }

        return maxProfit;
	}
