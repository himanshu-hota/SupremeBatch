public static void binarySearch(int arr[] , int target) {
		int n = arr.length;
		int start = 0;
		int end = n;
		int mid = start + (end - start)/2;
		while(start<=end) {
			// if element is found then break the loop
			if(arr[mid] == target) {
				
				System.out.println("Element found at " + (mid + 1) +  "th position ");
				return;
			}else if(target < arr[mid]) { // if target is less than middle element then search for it in left part of the array
				
				end = mid-1;
			}else { // if target is greater than middle element then search for it in right part of the array
				
				start = mid+1;
			}
			
			//update middle index
			mid =  start + (end - start)/2;
		}
		
		System.out.println("ELement not Found!!!");
	}
