 static void printSubArray(int arr[]){
        for(int start = 0; start < arr.length; start++){
            int end = start;
            printSubArrayRec(arr,start,end); // recursive call for each element
        }
    }

    static void printSubArrayRec(int arr[] , int start, int end){

        //base case
        if(end == arr.length){
            return;
        }
        
        // printing array elements
        for(int i = start ;i<=end ; i++){
            System.out.print(arr[i]);
        }

        // printing new line
        System.out.println();

        // recursive call
        printSubArrayRec(arr,start,end+1);

    }
