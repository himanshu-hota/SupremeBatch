    public int compress(char[] chars) {
        
        //total count of characters in out answer
        int total_count = 0;
        // grab current character
        char prev_char = chars[0];
        // counter for occurrence of character
        int occurrence = 0;
        // index on which characters and frequency will be stored
        int index = 0;

        // traverse array
        for(char ch : chars){ 
            // if current character is equal to the previous character then increase the frequency
            if(ch == prev_char) occurrence++;
            else{
                // if any character occurs more than once then store it and its frequency
                if(occurrence > 1){
                    // convert the frequency into string
                    String str = Integer.toString(occurrence);
                    // put character at index
                    chars[index++] = prev_char;
                    // increase the total count
                    total_count++;
                    // add the length of the digit of frequency into the array
                    total_count += str.length();
                    // add individual digit to the array
                    for(char c : str.toCharArray()) chars[index++] = c;
                }else{
                    // increase the total count 
                    total_count++;
                    // put the character at the index
                    chars[index++] = prev_char;
                }

                // update prev_char to current current character
                prev_char = ch;
                // reset occurrence to 1
                occurrence = 1;
            }
        }
        if(occurrence > 1){
            total_count++;
            chars[index++] = prev_char;
            String str = Integer.toString(occurrence);
            total_count += str.length();
            for(char c : str.toCharArray()) chars[index++] = c;
        }
        else{
            total_count++;
            chars[index++] = prev_char;
        }
        return total_count;
    }
